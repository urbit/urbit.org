::  Lesson 11: live todo list with feather-1
::
::  Install path: /learn/11-todo-live-feather-1
::  View URL:     /o/learn/11-todo-live-feather-1
::
::  New ideas:
::    `feather-1` splits a page into `get`, `post`, and `ui-diff`.
::    `session-sub:r` turns session changes into Datastar patches.
::
!:
%-  feather-1
=>
  |%
  +$  item  [id=@da text=@t]
  ::
  ++  read-items
    |=  req=http-request
    =/  m  (strand ,(list item))
    ^-  form:m
    =*  r  ~(. server req ~)
    ;<  old=(unit (list item))  bind:m  (get-session:r (list item))
    (pure:m (fall old ~))
  ::
  ++  list-section
    |=  items=(list item)
    ^-  manx
    ;section.fc.g3(id "todo-live-list")
      ;+  ?:  =(~ items)
            ;p.o5: no todos yet
          ;/("")
      ;ul
        ;*  %+  turn  items
            |=  it=item
            ^-  manx
            =/  sid=tape  (scow %da id.it)
            ;li.fr.ac.g2
              ;span.grow: {(trip text.it)}
              ;button.px2.py1.br2.hover.o6
                =data-on_click  "@post('?op=del&id={sid}')"
                ; delete
              ==
            ==
      ==
    ==
  --
|_  req=http-request
++  get
  =/  m  (strand ,get-response)
  ^-  form:m
  ;<  items=(list item)  bind:m  (read-items req)
  %-  pure:m
  :-  *marl
  ;main.page.prose
    =data-signals  "\{new: ''}"
    ;h1: Lesson 11 — live todo
    ;p
      ; This page updates from `%hawk499` session facts without a full reload.
    ==
    ;form.fr.g2.ac
      =data-on_submit  "@post('?op=add', \{contentType: 'form'}); $new = ''"
      ;input.grow(name "text", data-bind "new", placeholder "todo text");
      ;button(type "submit"): add
    ==
    ;+  (list-section items)
  ==
::
++  post
  =/  m  (strand ,~)
  ^-  form:m
  =*  r  ~(. server req ~)
  =/  op=@t  (fall (~(get by query.req) 'op') 'add')
  ;<  items=(list item)  bind:m  (read-items req)
  =/  next=(list item)
    ?+  op  items
      %add
        =/  form=(map @t @t)  formencoded-body:r
        =/  text=@t  (fix-newlines (fall (~(get by form) 'text') ''))
        ?:  =('' text)  items
        [[now.req text] items]
      %del
        =/  tid=@da  (slav %da (fall (~(get by query.req) 'id') '~0'))
        (skip items |=(it=item =(id.it tid)))
      %clear
        ~
    ==
  ;<  ~  bind:m  (set-session:r !>(next))
  (respond-patch:r ~)
::
++  ui-diff
  =*  r  ~(. server req ~)
  ^-  datastar-conduit
  :-  !>(~)
  :~  %-  session-sub:r
      |=  [state=vase ev=watch-sign]
      =/  m  (strand ,[vase (list datastar-event)])
      ^-  form:m
      ?.  ?=(%fact -.ev)  (pure:m [state ~])
      =/  items=(list item)  (fall ((soft (list item)) q.q.cage.ev) ~)
      (pure:m [state ~[[%outer ~ ~ (list-section items)]]])
  ==

      ==
    ==

|_  req=http-request
++  get
  =/  m  (strand ,get-response)
  ^-  form:m
  ;<  rows=(list @t)  bind:m  (read-log req)
  %-  pure:m
  :-  *marl
  ;main.page.prose
    =data-signals  "\{msg: '', saving: false}"

::
++  post
  =/  m  (strand ,~)
  ^-  form:m
  =*  r  ~(. server req ~)
  =/  op=@t  (fall (~(get by query.req) 'op') 'append')
  =/  form=(map @t @t)  formencoded-body:r
  =/  msg=@t  (fix-newlines (fall (~(get by form) 'msg') ''))
  ;<  rows=(list @t)  bind:m  (read-log req)
  =/  next=(list @t)
    ?:  =('clear' op)  ~
    ?:  =('' msg)  rows
    [msg rows]
  ;<  ~  bind:m  (set-session:r !>(next))
  (respond-patch:r ~)
::
++  ui-diff
  =*  r  ~(. server req ~)
  ^-  datastar-conduit
  :-  !>(~)
  :~  %-  session-sub:r
      |=  [state=vase ev=watch-sign]
      =/  m  (strand ,[vase (list datastar-event)])
      ^-  form:m
      ?.  ?=(%fact -.ev)  (pure:m [state ~])
      =/  rows=(list @t)  (fall ((soft (list @t)) q.q.cage.ev) ~)
      (pure:m [state ~[[%outer ~ ~ (log-section rows)]]])
  ==
